WLS

Perhitungan Weights

resid_abs <- abs(model$residuals)
fit <- lm(resid_abs ~ data$youtube)
data.weights <- 1 / fit$fitted.values^2

Pemodelan

model.lmw <- lm(sales~ youtube, 
              data = data, 
              weights = data.weights)
